Grid Count

Grid Count will generate Count/Index for Grid (Column and Row).
Methods that used for count is SetGridCount() and GetGridCount().

void insertFrame(M_help data, int row, int column)
{
    var frame = new Frame() { 
        Padding = 10,
    };

    frame.Content = new Label()
    {
        Text = data.judul
    };

    var tap = new TapGestureRecognizer();
    tap.Tapped += (s, e) => 
    {
        Navigation.PushAsync(new HelpDetail(data) { Title = data.judul });
    };

    frame.GestureRecognizers.Add(tap);

    grid.Children.Add(frame, row, column);
}
